```kotlin
object VoiceSelector {

    // Lista mantenida de voces masculinas conocidas (Google neural + comunes)
    private val KNOWN_MALE_IDS = setOf(
        "es-AR-Tomas-Neural", "es-AR-Tomas",
        "es-MX-Jorge-Neural", "es-MX-Jorge",
        "es-US-Neural2-B", "es-US-Neural2-C",
        "es-ES-Alvaro-Neural", "es-ES-Alvaro",
    )

    private val MALE_NAME_HINTS = setOf(
        "tomas", "jorge", "alvaro", "andres", "diego",
        "pablo", "carlos", "javier", "manuel", "miguel"
    )

    private val FEMALE_NAME_HINTS = setOf(
        "elena", "maria", "sofia", "lucia", "camila",
        "valentina", "ana", "paula", "laura"
    )

    fun selectBest(voices: Set<Voice>): Voice? {
        return voices
            .filter { it.locale.language == "es" }
            .sortedWith(
                compareByDescending<Voice> { it.scoreLocale() }
                    .thenByDescending { it.scoreGender() }     // male > unknown > female
                    .thenByDescending { it.isNeural() }
                    .thenByDescending { it.quality }
                    .thenBy { it.isNetworkConnectionRequired }
            )
            .firstOrNull()
    }

    private fun Voice.scoreLocale(): Int = when (locale.toLanguageTag()) {
        "es-AR" -> 100
        "es-419" -> 80
        "es-MX" -> 60
        "es-CL", "es-PE", "es-CO" -> 50
        "es-US" -> 40
        "es-ES" -> 30
        else -> 10
    }

    private fun Voice.scoreGender(): Int {
        if (name in KNOWN_MALE_IDS) return 100
        val lc = name.lowercase()
        if (MALE_NAME_HINTS.any { it in lc }) return 80
        if (FEMALE_NAME_HINTS.any { it in lc }) return 0
        return 50 // unknown â€” better than female, worse than male
    }

    private fun Voice.isNeural(): Boolean =
        name.contains("neural", ignoreCase = true) ||
        name.contains("wavenet", ignoreCase = true) ||
        features.contains("network-tts")
}
```

```kotlin
class PhraseGenerator(
    private val llm: LlmClient,
    private val fallbacks: PhraseFallbacks = PhraseFallbacks,
) {
    private val cache = LruCache<String, String>(100)
    private val mutex = Mutex()

    suspend fun forIntent(intent: PhraseIntent, params: PhraseParams): String {
        val key = key(intent, params)
        cache.get(key)?.let { return it }

        return mutex.withLock {
            cache.get(key)?.let { return it }
            val generated = generateOrFallback(intent, params)
            cache.put(key, generated)
            generated
        }
    }

    private suspend fun generateOrFallback(intent: PhraseIntent, params: PhraseParams): String {
        return runCatching {
            withTimeout(800) {
                val prompt = PromptBuilder.buildPhraseGenPrompt(intent, params)
                val raw = llm.generatePhrase(prompt)
                if (passesQualityChecks(raw)) raw else fallbacks.forIntent(intent, params)
            }
        }.getOrElse { fallbacks.forIntent(intent, params) }
    }

    private fun passesQualityChecks(s: String): Boolean {
        val words = s.split(Regex("\\s+")).count()
        if (words > 20 || words < 2) return false
        // Rechazo formal: "usted", "ustedes" â€” Beto es amigo, no asistente formal
        if (s.contains(Regex("\\busted(es)?\\b", RegexOption.IGNORE_CASE))) return false
        // Rechazo de "su" / "sus" en uso formal de tercera persona â€” heurĂ­stica simple
        // (acepta "su casa" si claramente posesivo de tercero, rechaza si dirigido al user)
        if (s.contains(Regex("\\b(su|sus)\\s+(mensaje|llamada|tel[eĂ©]fono|contacto|familia|nieto|hijo|hija)\\b", RegexOption.IGNORE_CASE))) return false
        if (s.contains("[DNI]") || s.contains("[TEL]") || s.contains("[TARJETA]")) return false
        return true
    }

    suspend fun warmCache() {
        // Pre-genera las frases comunes en background, no bloquea boot
        listOf(
            PhraseIntent.UNKNOWN_COMMAND to PhraseParams.empty(),
            PhraseIntent.FAILED_INTENT to PhraseParams.empty(),
        ).forEach { (intent, params) -> forIntent(intent, params) }
    }
}
```
